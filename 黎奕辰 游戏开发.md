using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;


public class Snake : MonoBehaviour
{
    
    private int y;
    private int x;
    public int n;
    public int footstep;
    public float speed = 0.3f;
    private Vector3 hp;
    public List<Transform> bodylengh = new List<Transform>();
    public GameObject bodypb;
    public Sprite[] bodysprites = new Sprite[2];
    public Sprite[] bodysprites2 = new Sprite[2];
    private Transform canvas;
    private bool godie = false;
 

    void Update()
    {
      
        if (Input.GetKey(KeyCode.W)&&y!=-footstep && NewBehaviourScript1.Again.Pause == false && godie == false)
        {
            gameObject.transform.localRotation = Quaternion.Euler(0, 0, 0);
            x = 0;
            y = footstep;
        }
        if (Input.GetKey(KeyCode.S)&&y!=footstep && NewBehaviourScript1.Again.Pause == false && godie == false)
        {
            gameObject.transform.localRotation = Quaternion.Euler(0, 0, 180);
            x = 0;
            y = -footstep;
        }
        if (Input.GetKey(KeyCode.A)&&x!=footstep && NewBehaviourScript1.Again.Pause == false && godie == false)
        {
            gameObject.transform.localRotation = Quaternion.Euler(0, 0, 90);
            y = 0;
            x = -footstep;
        }
        if (Input.GetKey(KeyCode.D)&&x!=-footstep && NewBehaviourScript1.Again.Pause == false && godie == false)
        {
            gameObject.transform.localRotation = Quaternion.Euler(0, 0,270 );
            y = 0;
            x = footstep;
        }
    }
    void Awake()
    {
        canvas = GameObject.Find("Canvas").transform;
    }
    void Grow()
    {
        int index = bodylengh.Count % 2;
        GameObject body = Instantiate(bodypb,new Vector3(2000,2000,0),Quaternion.identity);
        if (n == 0)
        {
            body.GetComponent<Image>().sprite = bodysprites[index];
        }
        else if(n==1)
        {
            body.GetComponent<Image>().sprite = bodysprites2[index];
        }
        body.transform.SetParent(canvas, false);
        bodylengh.Add(body.transform);
    }
    void Move()
    {
        hp = gameObject.transform.localPosition;
        gameObject.transform.localPosition = new Vector3(hp.x + x, hp.y + y, hp.z);
        if (bodylengh.Count > 0)
        {
            for(int i=bodylengh.Count-2;i>=0;i--)
            {
               bodylengh[i + 1].localPosition = bodylengh[i].localPosition;
           }
            bodylengh[0].localPosition = hp;
        }
    }
    void Start()
    {
        InvokeRepeating("Move", 0, speed);
        x = 0;
        y = footstep;
        Randoms();
        for (int i = 0; i < 2; i++)
        {
            Grow();
        }
    }
    private void OnTriggerEnter2D(Collider2D collision)
    {
        if(collision.tag=="food")
        {
            Destroy(collision.gameObject);
            NewBehaviourScript1.Again.UpdateUI();
            Grow();
            Increase();
            if (Random.Range(0,100)>85)
            {
                NewBehaviourScript.Again.Foodmaking(true);
            }
            else
            {
                NewBehaviourScript.Again.Foodmaking(false);
            }
        }
        else if(collision.tag=="goodluck")
        {
            Destroy(collision.gameObject);
            NewBehaviourScript1.Again.UpdateUI(Random.Range(10, 20));
            Grow();
            Increase();
        }
        else if(collision.tag == "body")
        {
            Death();
        }
        else
        {
            switch (collision.gameObject.name)
            {
                case "up":
                    {
                        transform.localPosition = new Vector3(transform.localPosition.x, -transform.localPosition.y + 30, transform.localPosition.z);
                        break;
                    }
                case "down":
                    {
                        transform.localPosition = new Vector3(transform.localPosition.x, -transform.localPosition.y - 30, transform.localPosition.z);
                        break;
                    }
                case "left":
                    {
                        transform.localPosition = new Vector3(-transform.localPosition.x+180, transform.localPosition.y , transform.localPosition.z);
                        break;
                    }
                case "right":
                    {
                        transform.localPosition = new Vector3(-transform.localPosition.x+230, transform.localPosition.y , transform.localPosition.z);
                        break;
                    }
            }
        }
    }
    void Death()
    {
        CancelInvoke();
        godie = true;
        StartCoroutine(Gameover(2f));
    }
    IEnumerator Gameover(float t)
    {
        yield return new WaitForSeconds(t);
        UnityEngine.SceneManagement.SceneManager.LoadScene(1);
    }
    void Increase()
    {
       if( NewBehaviourScript1.Again.score>=20&& NewBehaviourScript1.Again.score<50)
        {
            CancelInvoke();
            InvokeRepeating("Move", 0, speed = 0.20f);
        }
       else if(NewBehaviourScript1.Again.score>=50&& NewBehaviourScript1.Again.score<100)
        {
            CancelInvoke();
            InvokeRepeating("Move", 0, speed = 0.17f);
        }
       else if(NewBehaviourScript1.Again.score>=100&& NewBehaviourScript1.Again.score<150)
        {
            CancelInvoke();
            InvokeRepeating("Move", 0, speed = 0.15f);
        }
       else if(NewBehaviourScript1.Again.score>=150&& NewBehaviourScript1.Again.score<300)
        {
            CancelInvoke();
            InvokeRepeating("Move", 0, speed = 0.12f);
        }
       else if(NewBehaviourScript1.Again.score>=300)
        {
            CancelInvoke();
            InvokeRepeating("Move", 0, speed = 0.11f);
        }
    }
    void Randoms()
    {
        n = 1;
        if(Random.Range(1,100)>45)
        {
            n = 0;
        }
    }
}

using UnityEngine;
using UnityEngine.UI;


public class NewBehaviourScript : MonoBehaviour
{

    private static NewBehaviourScript _again;
    public int ylimit = 10;
    public int xlimit = 21;
    public int x2limit = 7;
    public Sprite[] foodsprites;
    public GameObject foodprefab;
    private Transform holdfood;
    public GameObject goodluckfb;
    public static NewBehaviourScript Again
    {
        get
        {
            return _again;
        }
    }

    void Awake()
    {
        _again = this;
    }

    public void Foodmaking(bool isluck)
    {
        int kind = Random.Range(0, foodsprites.Length);
        GameObject food = Instantiate(foodprefab);
        food.transform.SetParent(holdfood, false);
        food.GetComponent<Image>().sprite = foodsprites[kind];
        int x = Random.Range(-xlimit + x2limit, xlimit);
        int y = Random.Range(-ylimit, ylimit);
        food.transform.localPosition = new Vector3(x * 30, y * 30, 0);
        if(isluck)
        {
            GameObject goodluck = Instantiate(goodluckfb);
            goodluck.transform.SetParent(holdfood, false);
            x = Random.Range(-xlimit + x2limit, xlimit);
            y = Random.Range(-ylimit, ylimit);
            goodluck.transform.localPosition = new Vector3(x * 30, y * 30, 0);
        }
    }
    void Start()
    {
        holdfood = GameObject.Find("hoidfoodt").transform;
        Foodmaking(false);
    }
}
